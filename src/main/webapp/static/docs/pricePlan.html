取值定义
// 价格计划ID		 
PRICE_PLAN_ID = 230000000117;
// 价格计划名称
NAME = '美洲租机押金';
// 拼音简写
ALPHA = 'MZZJYJ';
// 创建人ID
CREATER = 44445;
费用类型(PRICE_PLAN_TYPE_CD)
# 字段值 描述
1 PRICE_PLAN_PKG_BUSI 手续费
2 PRICE_PLAN_DEDUCT_SCORE 积分扣减
3 PRICE_PLAN_AWARD_SCORE 积分赠送
4 PRICE_PLAN_DEDUCT_POINT 积点扣减
5 PRICE_PLAN_AWARD_POINT 积点赠送
6 PRICE_PLAN_PKG_PREPAY 预存
7 PRICE_PLAN_AWARD_PREPAY 赠送预存
8 PRICE_PLAN_PKG_DEPOSIT 押金
9 PRICE_PLAN_PKG_CARDFEE SIM卡费
10 PRICE_PLAN_PKG_BACK 退费
11 PRICE_PLAN_PKG_PUNISH 违约金
支付方式(PAY_METHOD)
# 字段值 描述
1 3 前台缴纳
2 4 预存抵扣
3 5 转账单
配置产品单元表(UP_PRODUCT_ITEM)
# 字段名 描述 取值
1 PRODUCT_ITEM_ID 产品单元ID PRICE_PLAN_ID
2 NAME 产品单元名称 NAME
3 ALPHA_NAME 首字母简写 ALPHA
4 ITEM_TYPE 产品单元类型 'PRICE_PLAN_PKG_BUSI'
5 CREATER 创建人 CREATER
6 STATE 产品单元状态 'U'
7 DEL_FLAG 删除标识 '1'
8 CREATE_DATE 创建时间 系统时间
配置属性表(UP_ATTR)
# 字段名 描述 取值
1 PRICE_PLAN_ID 价格计划ID PRICE_PLAN_ID
2 PRICE_PLAN_TYPE_CD 价格计划类型 'PRICE_PLAN_PKG_BUSI'
3 FEE_AMOUNT 资费费用(分) 350000
4 ACC_CODE 营业科目 5040251
5 CREATER 创建人 CREATER
6 CREATE_DATE 创建时间 系统时间
7 DEL_FLAG 删除标识 '1'
8 PAY_ACC_CODE 抵扣科目 5040251
9 PAY_METHOD 支付方式 5
10 CALC_FLAG 是否固定收取费用 0
